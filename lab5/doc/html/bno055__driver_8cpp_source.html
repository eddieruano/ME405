78 bool bno055_driver::initialize(void)
79 {
80  //uint8_t data_dump[1];
81  //ping the address to make sure we're in the right spot
82  if (p_i2c_driver -> ping(local_bno055_address))
83  {
84  *p_serial << "Successful Ping of BNO055" << endl;
85  }
86 
87  //now we set it to config mode
88  setMode(OPERATION_MODE_CONFIG);
89 
90  /* Reset */
91  writeRegister(BNO055_SYS_TRIGGER_ADDR, 0x20);
92 
93  while (!(p_i2c_driver -> ping(local_bno055_address)));
94  setMode(OPERATION_MODE_CONFIG);
95  // Config the BNO055
96  calibrate();
97  //Set Power to Normal
98  writeRegister(BNO055_PWR_MODE_ADDR, POWER_MODE_NORMAL);
99 
100  //Set the start Page Address
101  writeRegister(BNO055_PAGE_ID_ADDR, 0x00);
102 
103  //Mask the Triggers/Interrupts
104  writeRegister(BNO055_SYS_TRIGGER_ADDR, 0x00);
105 
106  /* Set the output units */
107 
108  uint8_t unitsel = (0 << 7) | // Orientation = Android
109  (0 << 4) | // Temperature = Celsius
110  (1 << 2) | // Euler = Degrees
111  (0 << 1) | // Gyro = Rads
112  (0 << 0); // Accelerometer = m/s^2
113  writeRegister(BNO055_UNIT_SEL_ADDR, unitsel);
114 
115 
116  setMode(OPERATION_MODE_NDOF);
117  systemStatus();
118  return true;
119 }

277 imu::Vector<3> bno055_driver::getVector(vector_type_t vector_type)
278 {
279  imu::Vector<3> xyz;
280 
281  //memset (data_dump, 0, 6);
282 
283  int16_t x, y, z;
284  x = y = z = 0;
285 
286  //Read vector data which is 6 bytes long
287  readRegister((bno055_reg_t)vector_type, data_dump, 6);
288 
289  x = ((int16_t)data_dump[0]) | (((int16_t)data_dump[1]) << 8);
290  y = ((int16_t)data_dump[2]) | (((int16_t)data_dump[3]) << 8);
291  z = ((int16_t)data_dump[4]) | (((int16_t)data_dump[5]) << 8);
292 
293  /* Convert the value to an appropriate range (section 3.6.4) */
294  /* and assign the value to the Vector type */
295  switch (vector_type)
296  {
297  case VECTOR_MAGNETOMETER:
298  /* 1uT = 16 LSB */
299  xyz[0] = ((double)x) / 16.0;
300  xyz[1] = ((double)y) / 16.0;
301  xyz[2] = ((double)z) / 16.0;
302  break;
303  case VECTOR_GYROSCOPE:
304  /* 1rps = 900 LSB */
305  xyz[0] = ((double)x) / 900.0;
306  xyz[1] = ((double)y) / 900.0;
307  xyz[2] = ((double)z) / 900.0;
308  break;
309  case VECTOR_EULER:
310  /* 1 degree = 16 LSB */
311  //Heading
312  xyz[0] = ((double)x) / 16.0;
313  //Roll
314  xyz[1] = ((double)y) / 16.0;
315  //Pitch
316  xyz[2] = ((double)z) / 16.0;
317  break;
318  case VECTOR_ACCELEROMETER:
319  case VECTOR_LINEARACCEL:
320  case VECTOR_GRAVITY:
321  /* 1m/s^2 = 100 LSB */
322  xyz[0] = ((double)x) / 100.0;
323  xyz[1] = ((double)y) / 100.0;
324  xyz[2] = ((double)z) / 100.0;
325  break;
326  }
327 
328  return xyz;
329 }

431 void bno055_driver::calibrate(void)
432 {
433  // Using these values to calibrate
434  // Accelerometer: 0 65515 17
435  // Gyro: 65535 65535 0
436  // Mag: 65520 65467 4
437  // Accel Radius: 1000
438  // Mag Radius: 71
439  uint8_t i;
440  uint8_t calibration_data[22];
441  memset(calibration_data, 0, 22);
442  uint16_t* Accel_X;
443  uint16_t* Accel_Y;
444  uint16_t* Accel_Z;
445  uint16_t* Gyro_X;
446  uint16_t* Gyro_Y;
447  uint16_t* Gyro_Z;
448  uint16_t* Mag_X;
449  uint16_t* Mag_Y;
450  uint16_t* Mag_Z;
451  uint16_t* Accel_Rad;
452  uint16_t* Mag_Rad;
453  *Accel_X = 0;
454  *Accel_Y = 65515;
455  *Accel_Z = 17;
456  *Gyro_X = 65535;
457  *Gyro_Y = 65535;
458  *Gyro_Z = 0;
459  *Mag_X = 65520;
460  *Mag_Y = 65467;
461  *Mag_Z = 4;
462  *Accel_Rad = 1000;
463  *Mag_Rad = 71;
464 
465 
466  /* Set Accel Offsets */
467  // Accel X
468  calibration_data[0] = (uint8_t)(LOW_MASK & *Accel_X);
469  calibration_data[1] = (uint8_t)((*Accel_X) >> 8);
470  // Accel Y
471  calibration_data[2] = (uint8_t)(LOW_MASK & *Accel_Y);
472  calibration_data[3] = (uint8_t)((*Accel_Y) >> 8);
473  // Accel Z
474  calibration_data[4] = (uint8_t)(LOW_MASK & *Accel_Z);
475  calibration_data[5] = (uint8_t)((*Accel_Z) >> 8);
476 
477  /* Set Gyro Offsets */
478  // Gyro X
479  calibration_data[6] = (uint8_t)(LOW_MASK & *Gyro_X);
480  calibration_data[7] = (uint8_t)((*Gyro_X) >> 8);
481  // Gyro Y
482  calibration_data[8] = (uint8_t)(LOW_MASK & *Gyro_Y);
483  calibration_data[9] = (uint8_t)((*Gyro_Y) >> 8);
484  // Gyro Z
485  calibration_data[10] = (uint8_t)(LOW_MASK & *Gyro_Z);
486  calibration_data[11] = (uint8_t)((*Gyro_Z) >> 8);
487 
488  /* Set Mag Offsets */
489  // Mag X
490  calibration_data[12] = (uint8_t)(LOW_MASK & *Mag_X);
491  calibration_data[13] = (uint8_t)((*Mag_X) >> 8);
492  // mag Y
493  calibration_data[14] = (uint8_t)(LOW_MASK & *Mag_Y);
494  calibration_data[15] = (uint8_t)((*Mag_Y) >> 8);
495  // Mag Z
496  calibration_data[16] = (uint8_t)(LOW_MASK & *Mag_Z);
497  calibration_data[17] = (uint8_t)((*Mag_Z) >> 8);
498 
499  /* Set Accel Radius Offset */
500  // Accel Radius LSB
501  calibration_data[18] = (uint8_t)(LOW_MASK & *Accel_Rad);
502  calibration_data[19] = (uint8_t)((*Accel_Rad) >> 8);
503 
504  /* Set Mag Radius Offset */
505  // Mag Radius LSB
506  calibration_data[20] = (uint8_t)(LOW_MASK & *Mag_Rad);
507  calibration_data[21] = (uint8_t)((*Mag_Rad) >> 8);
508 
509  for (i = 0; i <= 21; i++)
510  {
511  //*p_serial << "R0x" << hex << (i + OFFSET_REG_OFFSET) << ": " << calibration_data[i] << endl;
512  writeRegister((bno055_reg_t)(OFFSET_REG_OFFSET + i), calibration_data[i]);
513  }
514 
515 }
