106 /**
107  * @brief This method is called once by the RTOS scheduler.
108  * @details Each time that this method is run it initializes a tickcount and
109  * loops in an infinite for(;;) that evaluates the PID control loop
110  * and then sleeps for 10 milliseconds.
111  */
112 
113 void task_pid::run (void)
114 {
115  // Make a variable which will hold times to use for precise task scheduling
116  TickType_t previousTicks = xTaskGetTickCount ();
117 
118  for (;;)
119  {
120  int16_t ref = setpoint->get();
121  //*p_serial << PMS("SetPoint: ") << ref << endl;
122  int16_t act = feedback->get();
123  // proportional
124  int16_t err = ref - act;
125  // integral
126  err_sum = sssub( ssadd(err, err_sum), ssdiv(ssmul(KW, windup), 1024) );
127  // derivative
128  int16_t err_deriv = act - old_act;
129  old_act = act;
130  int16_t err_tot =
131  ssadd(
132  ssdiv(ssmul(KP,err),1024),
133  ssadd(
134  ssdiv(ssmul(KI,err_sum),1024),
135  ssdiv(ssmul(KD,err_deriv),1024)
136  )
137  );
138  int16_t out;
139  // saturation limits
140  if (err_tot > MAX)
141  {
142  out = MAX;
143  }
144  else if (err_tot < MIN)
145  {
146  out = MIN;
147  }
148  else out = err_tot;
149 
150  windup = sssub(err_tot, out);
151  motor_directive -> put(SETPOWER);
152  output->put(out);
153  // This is a method we use to cause a task to make one run through its task
154  // loop every N milliseconds and let other tasks run at other times
155  delay_from_for_ms (previousTicks, 1);
156  }
157 }
