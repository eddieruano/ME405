85 /**
86  * @brief Inititalize Pullups, Power Redux, Clock Prescalers, & Bitrates
87  *
88  * @details In order for us to get i2c comms going, we need to initialize
89  * a couple of registers. First we set the pull up resistors on
90  * PORTD, PINS[0,1] although it's not completely necessary since
91  * the ATmega1281 chip onboard controller for Two Wire Interface
92  * comms takes control of these lines, however for consistency
93  * and protection it's in the code. We also activate the two wire
94  * comms in the power reduction register and set no prescaler on
95  * the clock. Since no prescaler is declared the bitrate in TWBR
96  * can be low if the clock is high since we are trying to stay
97  * under 400,000 HZ total freq.
98  *
99  */
100 void i2c_driver::initialize(void)
101 {
102  /****** Begin initialization of I2C communication ******/
103  //Enable Internal Pull up on SDA and SCL pins [ATmega1281 PORTD PIN1 & PIN0]
104  PORTD |= (1 << SDA_PIN) | (1 << SCL_PIN);
105 
106  //Power Reduction Register 0 PPR0 (0x64)
107  //PPR0 -PRTWI | PRTIM2 | PRTIM0 | - | PRTIM1 | PRSPI | PRUSART0 | PRADC
108  //PPR0 ---7---|---6----|---5----|---4--|---3----|---2---|---1------|---0--
109  //PRTWI - (---Power Reduction Two Wire Input---)
110  //Normally set to 1, we need to set it to 0 to enable 2 wire comms
111 
112  PRR0 &= ~(1 << PRTWI);
113 
114  // For the next two registers consider this formula:
115  // SCL Freq = (CPU Internal Clock Frequency) / 16 + 2 (TWBR) * 4 ^ TWPS
116  // It sets the SCL
117 
118  // TWSR (TWI Status Register (Prescaler Bits)) -
119  // TWSR - TWS7 | TWS6 | TWS5 | TWS4 | TWS3 | â€“---- | TWPS1 | TWPS0
120  // TWSR - --7--|---6--|---5--|---4--|---3--|---2---|---1---|---0-----
121  // TWSR - (----------Status Bits-----------)-------(-Prescaler Bits-)
122  // TWSR will be 0x08 after a start condition
123  // The SCL clock should be 100kHz
124  //No prescaler so clean bottom two bits
125  TWSR &= ~((1 << TWPS1) | (1 << TWPS0));
126 
127  // TWBR (Bit Rate Generator Unit)
128  // TWBR (TWI Bit Rate Register) - will be 2 for a 50kHz SCL
129  // TWBR - TWBR7 | TWBR6 | TWBR5 | TWBR4 | TWBR3 | TWBR2 | TWBR1 | TWBR0
130  // TWBR - --7---|---6---|---5---|---4---|---3---|---2---|---1---|---0----
131  // Set bitrate to either 12 to make SCL got to 400kHz which is max
132  // OR set to 72 and get 100kHz on SCL
133  TWBR = 12;
134 
135  // Make sure we're still clear doe
136  TWCR &= ~(1 << TWIE);
137 
138  *p_serial << PMS("Initialized I2C: ") << endl;
139 }

321  * @brief Writes a payload of variable size to a device and target
322  * register using i2c protocol.
323  *
324  * @details Given a device address, target register, array of data, and
325  * length value, this function writes the data to the specified
326  * target using i2c protocol. It first takes the Mutex so no
327  * interfereence on the SDA line occurs while it is running the
328  * protocol. It saves the device address so that meaningful
329  * debuggning info can be relayed back to the caller. When it
330  * completes, it returns the Mutex and returns a boolean TRUE,
331  * otherwise FALSE.
332  *
333  * @param device 8bit Address of device we wish to establish comms with
334  *
335  * @param target 8bit Address of register we are reading on the device,
336  * it can also be the starting register of a multiple
337  * byte read.
338  *
339  * @param data Pointer to where we are storing the data read during
340  * the operation. It is assumed that the caller will not
341  * request more data than they are able to hold in this.
342  *
343  * @param length Number of bytes the caller wishes to read from the
344  * device, if a simple read operation is needed, this
345  * value should be 1. MAX bytes allowed to read is 255.
346  *
347  * @return Returns TRUE if the operation is successful, FALSE otherwise
348  */
349 bool i2c_driver::readData(uint8_t device, uint8_t target, uint8_t * data, uint8_t length)
350 {
351  // Update the local address holder, so we have a way to relay debugging information back to the caller
352  local_current_address = device;
353 
354  // initialize the loop counter
355  uint8_t count;
356 
357  //Takes the mutex, or waits for it based on portMAX_DELAY value established in Semph.h
358 
359  xSemaphoreTake (mutex, portMAX_DELAY);
360 
361  /*Step 1*/
362  // Start the protocol by master, first possible point of failure
363  if (!start()) RET_FAILURE
364 
365  /*Step 2*/
366  // Now we send Slave Address and wait for it to ack our request.
367  TWDR = DEVICE_WRITE;
368  clearEnable(STOP);
369  waitForSet();
370  // Check if 0x18 is returned by Slave
371  if (!checkError(TW_MT_SLA_ACK)) RET_FAILURE
372 
373  /*Step 3*/
374  //Send Register we want to access
375  TWDR = target;
376  clearEnable(STOP);
377  waitForSet();
378  // Check if it data request ack (0x28)
379  if (!checkError(TW_MT_DATA_ACK)) RET_FAILURE
380 
381  /*Step 4*/
382  // Repeat Start as per protocol requires
383  restart();
384 
385  /*Step 5*/
386  // Send Slave address of reg w/ READ command
387  TWDR = DEVICE_READ;
388  clearEnable(STOP);
389  waitForSet();
390  // Check if the read command is acknowledged, (0x40)
391  if (!checkError(TW_MR_SLA_ACK)) RET_FAILURE
392 
393  /*Step 6*/
394  // If we get to this point, we know that the next byte of data sent by the device will be data
395  clearEnable(CONTINUE);
396  waitForSet();
397  // Check if we have data waiting (0x48)
398  if (!checkError(TW_MR_DATA_ACK)) RET_FAILURE
399 
400  /*Step 7*/
401  // Here we set up a loop that will repeat this process for however many number of bytes required.
402  for (count = 0; count < length; count++)
403  {
404  data[count] = TWDR;
405  clearEnable(CONTINUE);
406  waitForSet();
407  if (!checkError(TW_MR_DATA_ACK)) RET_FAILURE
408  }
409 
410  /*Step 8*/
411  // Master sends a nack to stop the flow of data.
412  clearEnable(STOP);
413  waitForSet();
414  // Fail if the slave doesn't want to stop
415  if (!checkError(TW_MR_DATA_NACK)) RET_FAILURE
416 
417  /*Step 9*/
418  // Stop I2C Protocol
419  stop();
420  xSemaphoreGive (mutex);
421 
422  return true;
423 }
