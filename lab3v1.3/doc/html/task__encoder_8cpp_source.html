91 /**
92  * @brief This method is called once by the RTOS scheduler.
93  *
94  * @details Each time that this method is run it initializes a tickcount and
95  * loops in an infinite for(;;) loop that compares the last tick
96  * count stored locally with the most current reading of encoder
97  * count which lies in the global TaskShare variable
98  * 'encoder_count'. It then waits 10 ms to run again and since it
99  * has the highest priority we know that it will certainly run
100  * every 10ms. This means if we multiply the difference in encoder
101  * counts (local vs. global) and use the constant time difference
102  * we can obtain an estimate of the counts per sec.
103  */
104 
105 void task_encoder::run (void)
106 {
107  // Make a variable which will hold times to use for precise task scheduling
108  TickType_t previousTicks = xTaskGetTickCount ();
109 
110  // The loop to contunially run the motors
111  while (1)
112  {
113  //Subtract the last reading from this most current reading and multiply by 100 to obtain a per second reading.
114  count_per_sec -> put((encoder_count -> get() - last_count));
115  //rev/sec -> put(()*1000)
116 
117  // set the last count to the most current count to save it
118  last_count = encoder_count -> get();
119 
120  // Increment the run counter. This counter belongs to the parent class and can
121  // be printed out for debugging purposes
122  runs++;
123  // This is a method we use to cause a task to make one run through its task
