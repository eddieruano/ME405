145 
146  //* SETTING THE DATA DIRECTION *//
147 
148  // Need to set DDR to outputs for the input port
149  *input_DDR |= ((1 << input_APIN) | (1 << input_BPIN));
150  // Need to set the diagnostic as an (IN)
151  *diag_DDR &= ~(1 << diag_PIN);
152  // Need to set the Pulse Width Modulation Data Direction as an (OUT)
153  *pwm_DDR |= (1 << pwm_PIN);
154 
155  // Need to set the pull up resistor on diag port because i don't really know
156  *diag_PORT |= (1 << diag_PIN);
157 
158 
159  //** NOW SET LOGIC SO MOTOR WORKS W/ POTENTIOMETER **//
160  // signal > 0 = forwards!
161  // initialize a variable for speed
162  int16_t speed = 0;
163  if(sig >= 0)
164  {
165  // set_direction();
166  // PINA = HIGH, PINB = LOW
167  *input_PORT |= (1 << input_APIN);
168  *input_PORT &= ~(1 << input_BPIN);
169  // set_speed();
170  // put mod signal into speed reg (comparator)
171  // make sure value isn't negative
172  speed = sig;
173  if(speed < 0)
174  {
175  speed = speed * -1;
176  }
177 
178  *ocr_PORT = speed;

180  }

183  // set_direction();
