110  // Make a variable which will hold times to use for precise task scheduling

113  // Configure counter/timer 1 as a PWM for Motor Drivers.
114  // COM1A1/COM1B1 to set to non inverting mode.
115  // WGM10 to set to Fast PWM mode (only half ughhhhh)
116  TCCR1A |= (1 << WGM11) | (1 << WGM10) | (1 << COM1A1) | (1 << COM1B1);
117  // This is the second Timer/Counter Register
118  // WGM12 (other half of Fast PWM)
119  // CS11 Sets the presacler to 8 (010)
120  TCCR1B |= (1 << WGM12) | (1 << CS11);
121 
122  for (;;)
123  {
124  // creates local variable to reduce the number of calls to
125  // 'motor_directive'
126  uint8_t LOCAL_motor_directive = motor_directive -> get();
127 
128  //if Directive = 0, set the power of the motor.
129  if (LOCAL_motor_directive == SETPOWER)
130  {
131  motor -> set_power(motor_power->get());
132  }
133  //if Directive = 1, apply the motor brake
134  else if (LOCAL_motor_directive == BRAKE)
135  {
136  motor -> brake(motor_power ->get());
137  }
138  //if Directive = 2, freewheel the motor
139  else if (LOCAL_motor_directive == FREEWHEEL)
140  {
141  motor -> brake();
142  }
143  //if the Directive = 3, we enter potentiometer mode
144  else if (LOCAL_motor_directive == POTENTIOMETER)
145  {
146  //create variable to hold the reading of the adc. 1023
147  uint16_t duty_cycle = 2 * (p_adc -> read_once(0));
148  //convert the duty cycle into a signed variable and divide by
