00070 void task_user::run (void)
00071 {
00072     char char_in;                           // Character read from serial device
00073     time_stamp a_time;                      // Holds the time so it can be displayed
00074     uint32_t number_entered = 0;            // Holds a number being entered by user
00075 
00076     // Tell the user how to get into command mode (state 1), where the user interface
00077     // task does interesting things such as diagnostic printouts
00078     *p_serial << PMS ("Press 'h' or '?' for help") << endl;
00079 
00080     // This is an infinite loop; it runs until the power is turned off. There is one 
00081     // such loop inside the code for each task
00082     for (;;)
00083     {
00084         // Run the finite state machine. The variable 'state' is kept by parent class
00085         switch (state)
00086         {
00087             // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00088             // In state 0, we're in command mode, so when the user types characters, 
00089             // the characters are interpreted as commands to do something
00090             case (0):
00091                 if (p_serial->check_for_char ())            // If the user typed a
00092                 {                                           // character, read
00093                     char_in = p_serial->getchar ();         // the character
00094 
00095                     // In this switch statement, we respond to different characters as
00096                     // commands typed in by the user
00097                     switch (char_in)
00098                     {
00099                         // The 't' command asks what time it is right now
00100                         case ('t'):
00101                             *p_serial << (a_time.set_to_now ()) << endl;
00102                             break;
00103 
00104                         // The 's' command asks for version and status information
00105                         case ('s'):
00106                             show_status ();
00107                             break;
00108 
00109                         // The 'd' command has all the tasks dump their stacks
00110                         case ('d'):
00111                             print_task_stacks (p_serial);
00112                             break;
00113 
00114                         // The 'h' command is a plea for help; '?' works also
00115                         case ('h'):
00116                         case ('?'):
00117                             print_help_message ();
00118                             break;
00119 
00120                         // The 'n' command runs a test of entering a number
00121                         case ('n'):
00122                             *p_serial << PMS ("Enter decimal numeric digits, "
00123                                          "then RETURN or ESC") << endl;
00124                             number_entered = 0;
00125                             transition_to (1);
00126                             break;
00127 
00128                         // A control-C character causes the CPU to restart
00129                         case (3):
00130                             *p_serial << PMS ("Resetting AVR") << endl;
00131                             wdt_enable (WDTO_120MS);
00132                             for (;;);
00133                             break;
00134 
00135                         // If character isn't recognized, ask What's That Function?
00136                         default:
00137                             *p_serial << '"' << char_in << PMS ("\": WTF?") << endl;
00138                             break;
00139                     }; // End switch for characters
00140                 } // End if a character was received
00141 
00142                 break; // End of state 0
00143 
00144             // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00145             // In state 1, wait for user to enter digits and build 'em into a number
00146             case (1):
00147                 if (p_serial->check_for_char ())        // If the user typed a
00148                 {                                       // character, read
00149                     char_in = p_serial->getchar ();     // the character
00150 
00151                     // Respond to numeric characters, Enter or Esc only. Numbers are
00152                     // put into the numeric value we're building up
00153                     if (char_in >= '0' && char_in <= '9')
00154                     {
00155                         *p_serial << char_in;
00156                         number_entered *= 10;
00157                         number_entered += char_in - '0';
00158                     }
00159                     // Carriage return is ignored; the newline character ends the entry
00160                     else if (char_in == 10)
00161                     {
00162                         *p_serial << "\r";
00163                     }
00164                     // Carriage return or Escape ends numeric entry
00165                     else if (char_in == 13 || char_in == 27)
00166                     {
00167                         *p_serial << endl << PMS ("Number entered: ") 
00168                                   << number_entered << endl;
00169                         transition_to (0);
00170                     }
00171                     else
00172                     {
00173                         *p_serial << PMS ("<invalid char \"") << char_in 
00174                                   << PMS ("\">");
00175                     }
00176                 }
00177 
00178                 // Check the print queue to see if another task has sent this task 
00179                 // something to be printed
00180                 else if (p_print_ser_queue->check_for_char ())
00181                 {
00182                     p_serial->putchar (p_print_ser_queue->getchar ());
00183                 }
00184 
00185                 break; // End of state 1
00186 
00187             // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00188             // We should never get to the default state. If we do, complain and restart
00189             default:
00190                 *p_serial << PMS ("Illegal state! Resetting AVR") << endl;
00191                 wdt_enable (WDTO_120MS);
00192                 for (;;);
00193                 break;
00194 
00195         } // End switch state
00196 
00197         runs++;                             // Increment counter for debugging
00198 
00199         // No matter the state, wait for approximately a millisecond before we 
00200         // run the loop again. This gives lower priority tasks a chance to run
00201         delay_ms (1);
00202     }
00203 }
